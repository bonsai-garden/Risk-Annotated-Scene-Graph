277Look at this cropped object image and provide a single, concise label that describes what this object is.

279Guidelines:

318    def convert_numpy(self, obj):
319        """Recursively convert NumPy types to native Python types.
320
321        Args:
322            obj: Arbitrary object potentially containing NumPy types.
323
324        Returns:
325            Object with NumPy scalars/arrays converted to native types.
326        """
327        if isinstance(obj, dict):
328            return {k: self.convert_numpy(v) for k, v in obj.items()}
329        elif isinstance(obj, list):
330            return [self.convert_numpy(v) for v in obj]
331        elif isinstance(obj, np.integer):
332            return int(obj)
333        elif isinstance(obj, np.floating):
334            return float(obj)
335        elif isinstance(obj, np.ndarray):
336            return obj.tolist()
337        elif isinstance(obj, np.bool_):
338            return bool(obj)
339        else:
340            return obj

342    def strip_gt(self, obj):
343        """Recursively remove any gt fields and distance fields from a nested structure.
344
345        Args:
346            obj: Arbitrary object potentially containing gt fields.
347
348        Returns:
349            Object with gt fields removed.
350        """
351        if isinstance(obj, dict):
352            return {
353                k: self.strip_gt(v)
354                for k, v in obj.items()
355                if k not in {"gt", "distance", "confidence"}
356            }
357        if isinstance(obj, list):
358            return [self.strip_gt(v) for v in obj]
359        return obj

361    def remove_invalid_ids(self, graph, invalid_ids=None, expected_layers=None):
362        """Remove nodes and edges that reference invalid IDs or violate layer rules.
363
364        Args:
365            graph: Scene graph dictionary containing nodes/edges.
366            invalid_ids: Set of invalid ID values to remove.
367
368        Returns:
369            Cleaned graph dictionary.
370        """
371        if not isinstance(graph, dict):
372            return graph
373
374        invalid_ids = set(invalid_ids or set())
375        expected_layers = expected_layers or {}
376
377        def is_invalid(value):
378            if value in invalid_ids:
379                return True
380            try:
381                return int(value) in invalid_ids
382            except (TypeError, ValueError):
383                return False
384
385        nodes = graph.get("nodes")
386        removed_ids = set()
387        if isinstance(nodes, list):
388            filtered_nodes = []
389            for node in nodes:
390                if not isinstance(node, dict):
391                    filtered_nodes.append(node)
392                    continue
393
394                node_id = node.get("id")
395                if is_invalid(node_id):
396                    removed_ids.add(node_id)
397                    continue
398
399                name = node.get("name")
400                if isinstance(name, str) and name.strip() == "":
401                    removed_ids.add(node_id)
402                    continue
403
404                if node_id in expected_layers:
405                    expected = expected_layers.get(node_id)
406                    actual = node.get("layer")
407                    if isinstance(expected, str) and isinstance(actual, str):
408                        if expected.strip().upper() != actual.strip().upper():
409                            removed_ids.add(node_id)
410                            continue
411
412                filtered_nodes.append(node)
413            graph["nodes"] = filtered_nodes
414
415        edges = graph.get("edges")
416        if isinstance(edges, list):
417            filtered_edges = []
418            for edge in edges:
419                if isinstance(edge, dict):
420                    src = edge.get("from")
421                    dst = edge.get("to")
422                    if is_invalid(src) or is_invalid(dst):
423                        continue
424                    if src in removed_ids or dst in removed_ids:
425                        continue
426                filtered_edges.append(edge)
427            graph["edges"] = filtered_edges
428
429        return graph

362    def remove_invalid_ids(self, graph, invalid_ids=None, expected_layers=None):
363        """Remove nodes and edges that reference invalid IDs or violate layer rules.
364
365        Args:
366            graph: Scene graph dictionary containing nodes/edges.
367            invalid_ids: Set of invalid ID values to remove.
368
369        Returns:
370            Cleaned graph dictionary.
371        """
372        if not isinstance(graph, dict):
373            return graph
374
375        invalid_ids = set(invalid_ids or set())
376        expected_layers = expected_layers or {}
377
378        def is_invalid(value):
379            if value in invalid_ids:
380                return True
381            try:
382                return int(value) in invalid_ids
383            except (TypeError, ValueError):
384                return False
385
386        nodes = graph.get("nodes")
387        removed_ids = set()
388        if isinstance(nodes, list):
389            filtered_nodes = []
390            for node in nodes:
391                if not isinstance(node, dict):
392                    filtered_nodes.append(node)
393                    continue
394
395                node_id = node.get("id")
396                if is_invalid(node_id):
397                    removed_ids.add(node_id)
398                    continue
399
400                name = node.get("name")
401                if isinstance(name, str) and name.strip() == "":
402                    removed_ids.add(node_id)
403                    continue
404
405                if node_id in expected_layers:
406                    expected = expected_layers.get(node_id)
407                    actual = node.get("layer")
408                    if isinstance(expected, str) and isinstance(actual, str):
409                        if expected.strip().upper() != actual.strip().upper():
410                            removed_ids.add(node_id)
411                            continue
412
413                filtered_nodes.append(node)
414            graph["nodes"] = filtered_nodes
415
416        edges = graph.get("edges")
417        if isinstance(edges, list):
418            filtered_edges = []
419            for edge in edges:
420                if isinstance(edge, dict):
421                    src = edge.get("from")
422                    dst = edge.get("to")
423                    if is_invalid(src) or is_invalid(dst):
424                        continue
425                    if src in removed_ids or dst in removed_ids:
426                        continue
427                filtered_edges.append(edge)
428            graph["edges"] = filtered_edges
429
430        return graph

362    def remove_invalid_ids(self, graph, invalid_ids=None, expected_layers=None):
363        """Remove nodes and edges that reference invalid IDs or violate layer rules.
364
365        Args:
366            graph: Scene graph dictionary containing nodes/edges.
367            invalid_ids: Set of invalid ID values to remove.
368
369        Returns:
370            Cleaned graph dictionary.
371        """
372        if not isinstance(graph, dict):
373            return graph
374
375        invalid_ids = set(invalid_ids or set())
376        expected_layers = expected_layers or {}
377
378        def is_invalid(value):
379            if value in invalid_ids:
380                return True
381            try:
382                return int(value) in invalid_ids
383            except (TypeError, ValueError):
384                return False
385
386        nodes = graph.get("nodes")
387        removed_ids = set()
388        if isinstance(nodes, list):
389            filtered_nodes = []
390            for node in nodes:
391                if not isinstance(node, dict):
392                    filtered_nodes.append(node)
393                    continue
394
395                node_id = node.get("id")
396                if is_invalid(node_id):
397                    removed_ids.add(node_id)
398                    continue
399
400                name = node.get("name")
401                if isinstance(name, str) and name.strip() == "":
402                    removed_ids.add(node_id)
403                    continue
404
405                if node_id in expected_layers:
406                    expected = expected_layers.get(node_id)
407                    actual = node.get("layer")
408                    if isinstance(expected, str) and isinstance(actual, str):
409                        if expected.strip().upper() != actual.strip().upper():
410                            removed_ids.add(node_id)
411                            continue
412
413                filtered_nodes.append(node)
414            graph["nodes"] = filtered_nodes
415
416        edges = graph.get("edges")
417        if isinstance(edges, list):
418            filtered_edges = []
419            for edge in edges:
420                if isinstance(edge, dict):
421                    src = edge.get("from")
422                    dst = edge.get("to")
423                    if is_invalid(src) or is_invalid(dst):
424                        continue
425                    if src in removed_ids or dst in removed_ids:
426                        continue
427                filtered_edges.append(edge)
428            graph["edges"] = filtered_edges
429
430        return graph
