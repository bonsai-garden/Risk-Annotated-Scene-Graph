360    def parse_from_response(self, content: str) -> tuple[Any, bool]:
361        """Parse TOON content from a raw response string.
362
363        Args:
364            content: Raw response string.
365
366        Returns:
367            Tuple of parsed data and a success flag.
368        """
369        format_str = content.replace("\r\n", "\n").replace("\r", "\n")
370        if "```" in content:
371            start = format_str.find("```")
372            start = format_str.find("\n", start) + 1
373            end = format_str.find("```", start)
374            if end == -1:
375                format_str = format_str[start:]
376            else:
377                format_str = format_str[start:end]
378        else:
379            idx_nodes = format_str.find("nodes[")
380            idx_edges = format_str.find("edges[")
381            start_idx = min(i for i in [idx_nodes, idx_edges] if i != -1) if (idx_nodes != -1 or idx_edges != -1) else -1
382            if start_idx != -1:
383                format_str = format_str[start_idx:]
384
385        format_str = self._normalize_toon_counts(format_str)
386
387        try:
388            return self.loads(format_str.strip()), True
389        except:
390            try:
391                return self._decode(format_str.strip(), {"strict": False}), True
392            except:
393                try:
394                    return self.loads(content.strip()), True
395                except:
396                    return {"raw": format_str.strip()}, False
397
398    def _normalize_toon_counts(self, text: str) -> str:
399        lines = text.strip().splitlines()
400        if not lines:
401            return text
402
403        def find_header(prefix: str) -> int:
404            for i, line in enumerate(lines):
405                if line.lstrip().startswith(prefix + "["):
406                    return i
407            return -1
408
409        def count_rows(start_idx: int, end_idx: int) -> int:
410            count = 0
411            for line in lines[start_idx + 1:end_idx]:
412                if not line.strip():
413                    continue
414                if line.lstrip().startswith("nodes[") or line.lstrip().startswith("edges["):
415                    continue
416                count += 1
417            return count
418
419        nodes_idx = find_header("nodes")
420        edges_idx = find_header("edges")
421
422        if nodes_idx != -1:
423            nodes_end = edges_idx if edges_idx != -1 else len(lines)
424            nodes_count = count_rows(nodes_idx, nodes_end)
425            lines[nodes_idx] = re.sub(
426                r"^(\s*nodes)\[\d+\](.*)$",
427                rf"\1[{nodes_count}]\2",
428                lines[nodes_idx],
429            )
430
431        if edges_idx != -1:
432            edges_end = len(lines)
433            edges_count = count_rows(edges_idx, edges_end)
434            lines[edges_idx] = re.sub(
435                r"^(\s*edges)\[\d+\](.*)$",
436                rf"\1[{edges_count}]\2",
437                lines[edges_idx],
438            )
439
440        return "\n".join(lines)

444        """Get the TOON file extension.
445
446        Args:
447            None
448
449        Returns:
450            ".toon".
451        """
452        return '.toon'
