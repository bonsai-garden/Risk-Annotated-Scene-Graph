 86    def _update_robot_proximity_edges(self, max_objects: int = 3):
 87        """Link robot to nearest room and up to N nearest objects.
 88
 89        Args:
 90            max_objects: Max number of nearest objects to connect.
 91
 92        Returns:
 93            None
 94        """
 95        if self.robot_id not in self.scene_graph:
 96            return
 97
 98        robot_pos = self._parse_pos_if_string(self.scene_graph.nodes[self.robot_id].get("pos"))
 99        if robot_pos is None:
100            return
101
102        robot_arr = np.asarray(robot_pos, dtype=float)
103        if not np.isfinite(robot_arr).all():
104            return
105
106        # Remove previous proximity edges from robot to keep them updated each frame.
107        for _, dst in list(self.scene_graph.out_edges(self.robot_id)):
108            self.scene_graph.remove_edge(self.robot_id, dst)
109
110        room_candidates = []
111        object_candidates = []
112
113        for node_id, attrs in self.scene_graph.nodes(data=True):
114            if node_id == self.robot_id:
115                continue
116
117            layer = NodeLayer.from_string(attrs.get("layer", NodeLayer.OBJECT.name))
118            if layer not in {NodeLayer.ROOM, NodeLayer.OBJECT}:
119                continue
120
121            pos = self._parse_pos_if_string(attrs.get("pos"))
122            if pos is None:
123                continue
124
125            node_arr = np.asarray(pos, dtype=float)
126            if not np.isfinite(node_arr).all():
127                continue
128
129            distance = float(np.linalg.norm(node_arr - robot_arr))
130            if layer == NodeLayer.ROOM:
131                room_candidates.append((distance, node_id))
132            else:
133                object_candidates.append((distance, node_id))
134
135        room_candidates.sort(key=lambda item: item[0])
136        object_candidates.sort(key=lambda item: item[0])
137
138        if room_candidates:
139            dist, room_id = room_candidates[0]
140            self.scene_graph.add_edge(self.robot_id, room_id, rel="in", distance=round(dist, 4))
141
142        for dist, obj_id in object_candidates[:max_objects]:
143            self.scene_graph.add_edge(self.robot_id, obj_id, rel="near", distance=round(dist, 4))

168    def _calculate_edge_distance(self, src_id, dst_id):
169        """Compute Euclidean distance between two nodes, if possible.
170
171        Args:
172            src_id: Source node ID.
173            dst_id: Destination node ID.
174
175        Returns:
176            Distance in meters rounded to 4 decimals, or None.
177        """
178        try:
179            src_node = self.scene_graph.nodes.get(src_id)
180            dst_node = self.scene_graph.nodes.get(dst_id)
181            
182            if src_node is None or dst_node is None:
183                return None
184            
185            src_pos = self._parse_pos_if_string(src_node.get("pos"))
186            dst_pos = self._parse_pos_if_string(dst_node.get("pos"))
187            
188            if src_pos is None or dst_pos is None:
189                return None
190            
191            # Euclidean distance
192            src_arr = np.asarray(src_pos, dtype=float)
193            dst_arr = np.asarray(dst_pos, dtype=float)
194
195            if not np.isfinite(src_arr).all() or not np.isfinite(dst_arr).all():
196                return None
197            
198            distance = float(np.linalg.norm(dst_arr - src_arr))
199            return round(distance, 4)
200        except Exception as e:
201            return None

214    def process_vlm_update(self, cutout_before, vlm_output):
215        """Merge VLM output into the graph with appearance tracking.
216
217        Args:
218            cutout_before: Graph cutout sent to VLM.
219            vlm_output: Parsed VLM output graph.
220
221        Returns:
222            None
223        """
224        # Get IDs of nodes in VLM output
225        vlm_node_ids = set()
226        for node in vlm_output.get("nodes", []):
227            node_id = node["id"]
228            vlm_node_ids.add(node_id)
229            
230            # Increment appearance for detected objects
231            if node_id in self.scene_graph:
232                current_count = self.scene_graph.nodes[node_id].get("appearance", 0)
233                self.scene_graph.nodes[node_id]["appearance"] = current_count + 1
234                self.scene_graph.nodes[node_id].update(node)
235            else:
236                node_copy = node.copy()
237                node_copy["appearance"] = 1
238                if "layer" not in node_copy:
239                    node_copy["layer"] = NodeLayer.OBJECT.name
240                else:
241                    layer = NodeLayer.from_string(node_copy.get("layer"))
242                    if layer:
243                        node_copy["layer"] = layer.name
244                self.scene_graph.add_node(node_id, **node_copy)
245        
246        # Get IDs of nodes in cutout (reference for what VLM saw)
247        cutout_node_ids = set()
248        for node in cutout_before.get("nodes", []):
249            node_id = node["id"]
250            cutout_node_ids.add(node_id)
251        # Decrease appearance for objects that were in cutout but NOT in VLM output
252        for node_id in cutout_node_ids:
253            if node_id not in vlm_node_ids:
254                if node_id in self.scene_graph:
255                    current_count = self.scene_graph.nodes[node_id].get("appearance", 1)
256                    new_count = current_count - 1
257                    
258                    if new_count <= appearance_threshold:
259                        self.scene_graph.remove_node(node_id)
260                    else:
261                        self.scene_graph.nodes[node_id]["appearance"] = new_count
262        
263        # Update edges with appearance tracking
264        vlm_edge_set = set()
265        for edge in vlm_output.get("edges", []):
266            src, dst = edge["from"], edge["to"]
267
268            if src in self.scene_graph and dst in self.scene_graph:
269                vlm_edge_set.add((src, dst))
270                
271                rel = edge.get("rel", "")
272                distance = self._calculate_edge_distance(src, dst)
273                
274                if self.scene_graph.has_edge(src, dst):
275                    # Increment appearance for existing edge
276                    current_count = self.scene_graph[src][dst].get("appearance", 0)
277                    self.scene_graph[src][dst]["appearance"] = current_count + 1
278                    self.scene_graph[src][dst]["rel"] = rel
279                    self.scene_graph[src][dst]["distance"] = distance
280                else:
281                    # New edge with appearance = 1
282                    self.scene_graph.add_edge(src, dst, rel=rel, distance=distance, appearance=1)
283        
284        # Get edges from cutout (what VLM saw)
285        cutout_edge_set = set()
286        for edge in cutout_before.get("edges", []):
287            src, dst = edge["from"], edge["to"]
288            cutout_edge_set.add((src, dst))
289
290        # Decrease appearance for edges in cutout but NOT in VLM output
291        for src, dst in cutout_edge_set:
292            if (src, dst) not in vlm_edge_set:
293                if self.scene_graph.has_edge(src, dst):
294                    current_count = self.scene_graph[src][dst].get("appearance", 1)
295                    new_count = current_count - 1
296                    
297                    if new_count <= appearance_threshold:
298                        self.scene_graph.remove_edge(src, dst)
299                    else:
300                        self.scene_graph[src][dst]["appearance"] = new_count

382    def get_cutouts(self, tx, rotM, depth_img=None, depth_tolerance=0.5):
383        """Collect visible nodes/edges for the current camera pose.
384
385        Args:
386            tx: Translation vector.
387            rotM: Rotation matrix.
388            depth_img: Optional depth image in meters for frustum cutoff.
389            depth_tolerance: Extra range beyond depth values (meters).
390
391        Returns:
392            Dictionary with visible "nodes" and "edges".
393        """
394        visible_objects = {"nodes": [], "edges": []}
395        edge_candidates = []
396        visible_node_ids = set()
397        room_candidates = []
398        tx_arr = np.asarray(tx, dtype=float)
399
400        for node_id, attr in self.scene_graph.nodes(data=True):
401            # Skip ROBOT layer nodes from being sent to VLM
402            layer_raw = attr.get("layer", NodeLayer.OBJECT.name)
403            layer_enum = NodeLayer.from_string(layer_raw) or NodeLayer.OBJECT
404            layer = layer_enum.name
405            if layer == NodeLayer.ROBOT.name:
406                continue
407            
408            pos = self._parse_pos_if_string(attr.get("pos"))
409            if pos is None:
410                continue
411
412            point = np.asarray(pos, dtype=float)
413            if layer == NodeLayer.ROOM.name and np.isfinite(point).all():
414                distance = float(np.linalg.norm(point - tx_arr))
415                room_candidates.append((distance, node_id, attr, pos))
416
417            inside_cut = self.is_probably_visible(point, tx_arr, rotM, depth_img, depth_tolerance)
418            if inside_cut:
419                edge_candidates.append(node_id)
420                visible_objects["nodes"].append({
421                    "id": node_id,
422                    "name": attr.get("name", str(node_id)),
423                    "risk": attr.get("risk", ""),
424                    "pos": pos, 
425                    "confidence": attr.get("confidence", 0.0),
426                    "gt": bool(attr.get("gt", True)),
427                    "layer": layer
428                })
429                visible_node_ids.add(node_id)
430
431        if room_candidates:
432            room_candidates.sort(key=lambda item: item[0])
433            _, room_id, room_attr, room_pos = room_candidates[0]
434            if room_id not in visible_node_ids:
435                visible_objects["nodes"].append({
436                    "id": room_id,
437                    "name": room_attr.get("name", str(room_id)),
438                    "risk": room_attr.get("risk", ""),
439                    "pos": room_pos,
440                    "confidence": room_attr.get("confidence", 0.0),
441                    "gt": False,
442                    "layer": NodeLayer.ROOM.name
443                })
444                edge_candidates.append(room_id)
445                visible_node_ids.add(room_id)
446
447        for src, dst, edata in self.scene_graph.edges(data=True):
448            if src in edge_candidates and dst in edge_candidates:
449                visible_objects["edges"].append({
450                    "from": src, "to": dst, "rel": edata.get("rel", ""), "distance": edata.get("distance")
451                })
452
453        return visible_objects

455    def is_probably_visible(self, point, origin, rotM, depth_img=None, depth_tolerance=0.1):
456        """Check if a 3D point is within the camera's field of view and range.
457
458        Args:
459            point: 3D point in world coordinates.
460            origin: Camera origin in world coordinates.
461            rotM: Rotation matrix (camera to world).
462            depth_img: Optional depth image in meters for range cutoff.
463            depth_tolerance: Extra range beyond depth values (meters).
464
465        Returns:
466            True if point is likely visible, else False.
467        """
468        point = np.asarray(point, dtype=float)
469        origin = np.asarray(origin, dtype=float)
470
471        fx, fy = K_rgb[0, 0], K_rgb[1, 1]
472        height, width = img_size  
473
474        fov_x = 2 * np.degrees(np.arctan(width / (2 * fx)))
475        fov_y = 2 * np.degrees(np.arctan(height / (2 * fy)))
476
477        local = rotM.T @ (point - origin)
478        if local[2] <= 0:
479            return False
480
481        angle_x = np.degrees(np.arctan2(abs(local[0]), local[2]))
482        angle_y = np.degrees(np.arctan2(abs(local[1]), local[2]))
483        in_fov = (angle_x < (fov_x / 2) * margin) and (angle_y < (fov_y / 2) * margin)
484        if not in_fov:
485            return False
486
487        if depth_img is None:
488            return True
489
490        px = self.pos_calculator.pos_to_pixel(point[0], point[1], point[2], rotM, origin)
491        px_x, px_y = int(px[0]), int(px[1])
492        if px_x < 0 or px_y < 0:
493            return False
494
495        height, width = depth_img.shape[:2]
496        if px_x >= width or px_y >= height:
497            return False
498
499        depth_val = float(depth_img[px_y, px_x])
500        if depth_val <= 0 or np.isnan(depth_val):
501            return True
502
503        return local[2] <= depth_val + float(depth_tolerance)

505    def convert_graph_pos_to_pixel(self, graph, tx, rotM):
506        """Convert node 3D positions to normalized pixel coordinates.
507
508        Args:
509            graph: Graph dictionary with node positions.
510            tx: Translation vector.
511            rotM: Rotation matrix.
512
513        Returns:
514            Updated graph dictionary with x/y fields.
515        """
516        if graph is None:
517            return {"nodes": [], "edges": []}
518
519        height, width = img_size
520
521        for node in graph["nodes"]:
522            pos = node["pos"]
523            px = self.pos_calculator.pos_to_pixel(pos[0], pos[1], pos[2], rotM, tx)
524            x_pix, y_pix = float(px[0]), float(px[1])
525
526            # Clamp to pixel bounds
527            x_clamped = max(0, min(x_pix, width - 1))
528            y_clamped = max(0, min(y_pix, height - 1))
529
530            # GT: true if no clamping occurred (compare in pixel space)
531            gt_flag = (x_pix == x_clamped) and (y_pix == y_clamped)
532
533            # Convert to relative [0,1] and limit precision for VLM input
534            x_rel = round(x_clamped / (width - 1), 2)
535            y_rel = round(y_clamped / (height - 1), 2)
536
537            del node["pos"]
538            node["x"] = x_rel
539            node["y"] = y_rel
540            node["gt"] = gt_flag
541
542        return graph

544    def convert_graph_pixel_to_pos(self, graph, depth_img, tx, rotM):
545        """Convert node pixel coordinates to 3D positions using depth.
546
547        Args:
548            graph: Graph dictionary with node pixel coords.
549            depth_img: Depth image in meters.
550            tx: Translation vector.
551            rotM: Rotation matrix.
552
553        Returns:
554            Updated graph dictionary with pos fields.
555        """
556        height, width = img_size
557
558        for node in graph["nodes"]:
559            x_rel = node.get("x")
560            y_rel = node.get("y")
561
562            x_pix = int(round(x_rel * (width - 1)))
563            y_pix = int(round(y_rel * (height - 1)))
564
565            x_clipped = max(0, min(x_pix, width - 1))
566            y_clipped = max(0, min(y_pix, height - 1))
567
568            if x_pix != x_clipped or y_pix != y_clipped:
569                node["gt"] = False
570
571            pos = self.pos_calculator.pixel_to_pos(x_clipped, y_clipped, depth_img, rotM, tx)
572
573            del node["x"]
574            del node["y"]
575            node["pos"] = pos
576
577        return graph

 87    def _update_robot_proximity_edges(self, max_objects: int = 3):
 88        """Link robot to nearest room and up to N nearest objects.
 89
 90        Args:
 91            max_objects: Max number of nearest objects to connect.
 92
 93        Returns:
 94            None
 95        """
 96        if self.robot_id not in self.scene_graph:
 97            return
 98
 99        robot_pos = self._parse_pos_if_string(self.scene_graph.nodes[self.robot_id].get("pos"))
100        if robot_pos is None:
101            return
102
103        robot_arr = np.asarray(robot_pos, dtype=float)
104        if not np.isfinite(robot_arr).all():
105            return
106
107        # Remove previous proximity edges from robot to keep them updated each frame.
108        for _, dst in list(self.scene_graph.out_edges(self.robot_id)):
109            self.scene_graph.remove_edge(self.robot_id, dst)
110
111        room_candidates = []
112        object_candidates = []
113
114        for node_id, attrs in self.scene_graph.nodes(data=True):
115            if node_id == self.robot_id:
116                continue
117
118            layer = NodeLayer.from_string(attrs.get("layer", NodeLayer.OBJECT.name))
119            if layer not in {NodeLayer.ROOM, NodeLayer.OBJECT}:
120                continue
121
122            pos = self._parse_pos_if_string(attrs.get("pos"))
123            if pos is None:
124                continue
125
126            node_arr = np.asarray(pos, dtype=float)
127            if not np.isfinite(node_arr).all():
128                continue
129
130            distance = float(np.linalg.norm(node_arr - robot_arr))
131            if layer == NodeLayer.ROOM:
132                room_candidates.append((distance, node_id))
133            else:
134                object_candidates.append((distance, node_id))
135
136        room_candidates.sort(key=lambda item: item[0])
137        object_candidates.sort(key=lambda item: item[0])
138
139        if room_candidates:
140            dist, room_id = room_candidates[0]
141            self.scene_graph.add_edge(self.robot_id, room_id, rel="in", distance=round(dist, 4))
142
143        for dist, obj_id in object_candidates[:max_objects]:
144            self.scene_graph.add_edge(self.robot_id, obj_id, rel="near", distance=round(dist, 4))

169    def _calculate_edge_distance(self, src_id, dst_id):
170        """Compute Euclidean distance between two nodes, if possible.
171
172        Args:
173            src_id: Source node ID.
174            dst_id: Destination node ID.
175
176        Returns:
177            Distance in meters rounded to 4 decimals, or None.
178        """
179        try:
180            src_node = self.scene_graph.nodes.get(src_id)
181            dst_node = self.scene_graph.nodes.get(dst_id)
182            
183            if src_node is None or dst_node is None:
184                return None
185            
186            src_pos = self._parse_pos_if_string(src_node.get("pos"))
187            dst_pos = self._parse_pos_if_string(dst_node.get("pos"))
188            
189            if src_pos is None or dst_pos is None:
190                return None
191            
192            # Euclidean distance
193            src_arr = np.asarray(src_pos, dtype=float)
194            dst_arr = np.asarray(dst_pos, dtype=float)
195
196            if not np.isfinite(src_arr).all() or not np.isfinite(dst_arr).all():
197                return None
198            
199            distance = float(np.linalg.norm(dst_arr - src_arr))
200            return round(distance, 4)
201        except Exception as e:
202            return None

215    def process_vlm_update(self, cutout_before, vlm_output):
216        """Merge VLM output into the graph with appearance tracking.
217
218        Args:
219            cutout_before: Graph cutout sent to VLM.
220            vlm_output: Parsed VLM output graph.
221
222        Returns:
223            None
224        """
225        # Get IDs of nodes in VLM output
226        vlm_node_ids = set()
227        for node in vlm_output.get("nodes", []):
228            node_id = node["id"]
229            vlm_node_ids.add(node_id)
230            
231            # Increment appearance for detected objects
232            if node_id in self.scene_graph:
233                current_count = self.scene_graph.nodes[node_id].get("appearance", 0)
234                self.scene_graph.nodes[node_id]["appearance"] = current_count + 1
235                self.scene_graph.nodes[node_id].update(node)
236            else:
237                node_copy = node.copy()
238                node_copy["appearance"] = 1
239                if "layer" not in node_copy:
240                    node_copy["layer"] = NodeLayer.OBJECT.name
241                else:
242                    layer = NodeLayer.from_string(node_copy.get("layer"))
243                    if layer:
244                        node_copy["layer"] = layer.name
245                self.scene_graph.add_node(node_id, **node_copy)
246        
247        # Get IDs of nodes in cutout (reference for what VLM saw)
248        cutout_node_ids = set()
249        for node in cutout_before.get("nodes", []):
250            node_id = node["id"]
251            cutout_node_ids.add(node_id)
252        # Decrease appearance for objects that were in cutout but NOT in VLM output
253        for node_id in cutout_node_ids:
254            if node_id not in vlm_node_ids:
255                if node_id in self.scene_graph:
256                    current_count = self.scene_graph.nodes[node_id].get("appearance", 1)
257                    new_count = current_count - 1
258                    
259                    if new_count <= appearance_threshold:
260                        self.scene_graph.remove_node(node_id)
261                    else:
262                        self.scene_graph.nodes[node_id]["appearance"] = new_count
263        
264        # Update edges with appearance tracking
265        vlm_edge_set = set()
266        for edge in vlm_output.get("edges", []):
267            src, dst = edge["from"], edge["to"]
268
269            if src in self.scene_graph and dst in self.scene_graph:
270                vlm_edge_set.add((src, dst))
271                
272                rel = edge.get("rel", "")
273                distance = self._calculate_edge_distance(src, dst)
274                
275                if self.scene_graph.has_edge(src, dst):
276                    # Increment appearance for existing edge
277                    current_count = self.scene_graph[src][dst].get("appearance", 0)
278                    self.scene_graph[src][dst]["appearance"] = current_count + 1
279                    self.scene_graph[src][dst]["rel"] = rel
280                    self.scene_graph[src][dst]["distance"] = distance
281                else:
282                    # New edge with appearance = 1
283                    self.scene_graph.add_edge(src, dst, rel=rel, distance=distance, appearance=1)
284        
285        # Get edges from cutout (what VLM saw)
286        cutout_edge_set = set()
287        for edge in cutout_before.get("edges", []):
288            src, dst = edge["from"], edge["to"]
289            cutout_edge_set.add((src, dst))
290
291        # Decrease appearance for edges in cutout but NOT in VLM output
292        for src, dst in cutout_edge_set:
293            if (src, dst) not in vlm_edge_set:
294                if self.scene_graph.has_edge(src, dst):
295                    current_count = self.scene_graph[src][dst].get("appearance", 1)
296                    new_count = current_count - 1
297                    
298                    if new_count <= appearance_threshold:
299                        self.scene_graph.remove_edge(src, dst)
300                    else:
301                        self.scene_graph[src][dst]["appearance"] = new_count

383    def get_cutouts(self, tx, rotM, depth_img=None, depth_tolerance=0.5):
384        """Collect visible nodes/edges for the current camera pose.
385
386        Args:
387            tx: Translation vector.
388            rotM: Rotation matrix.
389            depth_img: Optional depth image in meters for frustum cutoff.
390            depth_tolerance: Extra range beyond depth values (meters).
391
392        Returns:
393            Dictionary with visible "nodes" and "edges".
394        """
395        visible_objects = {"nodes": [], "edges": []}
396        edge_candidates = []
397        visible_node_ids = set()
398        room_candidates = []
399        tx_arr = np.asarray(tx, dtype=float)
400
401        for node_id, attr in self.scene_graph.nodes(data=True):
402            # Skip ROBOT layer nodes from being sent to VLM
403            layer_raw = attr.get("layer", NodeLayer.OBJECT.name)
404            layer_enum = NodeLayer.from_string(layer_raw) or NodeLayer.OBJECT
405            layer = layer_enum.name
406            if layer == NodeLayer.ROBOT.name:
407                continue
408            
409            pos = self._parse_pos_if_string(attr.get("pos"))
410            if pos is None:
411                continue
412
413            point = np.asarray(pos, dtype=float)
414            if layer == NodeLayer.ROOM.name and np.isfinite(point).all():
415                distance = float(np.linalg.norm(point - tx_arr))
416                room_candidates.append((distance, node_id, attr, pos))
417
418            inside_cut = self.is_probably_visible(point, tx_arr, rotM, depth_img, depth_tolerance)
419            if inside_cut:
420                edge_candidates.append(node_id)
421                visible_objects["nodes"].append({
422                    "id": node_id,
423                    "name": attr.get("name", str(node_id)),
424                    "risk": attr.get("risk", ""),
425                    "pos": pos, 
426                    "confidence": attr.get("confidence", 0.0),
427                    "gt": bool(attr.get("gt", True)),
428                    "layer": layer
429                })
430                visible_node_ids.add(node_id)
431
432        if room_candidates:
433            room_candidates.sort(key=lambda item: item[0])
434            _, room_id, room_attr, room_pos = room_candidates[0]
435            if room_id not in visible_node_ids:
436                visible_objects["nodes"].append({
437                    "id": room_id,
438                    "name": room_attr.get("name", str(room_id)),
439                    "risk": room_attr.get("risk", ""),
440                    "pos": room_pos,
441                    "confidence": room_attr.get("confidence", 0.0),
442                    "gt": False,
443                    "layer": NodeLayer.ROOM.name
444                })
445                edge_candidates.append(room_id)
446                visible_node_ids.add(room_id)
447
448        for src, dst, edata in self.scene_graph.edges(data=True):
449            if src in edge_candidates and dst in edge_candidates:
450                visible_objects["edges"].append({
451                    "from": src, "to": dst, "rel": edata.get("rel", ""), "distance": edata.get("distance")
452                })
453
454        return visible_objects

456    def is_probably_visible(self, point, origin, rotM, depth_img=None, depth_tolerance=0.1):
457        """Check if a 3D point is within the camera's field of view and range.
458
459        Args:
460            point: 3D point in world coordinates.
461            origin: Camera origin in world coordinates.
462            rotM: Rotation matrix (camera to world).
463            depth_img: Optional depth image in meters for range cutoff.
464            depth_tolerance: Extra range beyond depth values (meters).
465
466        Returns:
467            True if point is likely visible, else False.
468        """
469        point = np.asarray(point, dtype=float)
470        origin = np.asarray(origin, dtype=float)
471
472        fx, fy = K_rgb[0, 0], K_rgb[1, 1]
473        height, width = img_size  
474
475        fov_x = 2 * np.degrees(np.arctan(width / (2 * fx)))
476        fov_y = 2 * np.degrees(np.arctan(height / (2 * fy)))
477
478        local = rotM.T @ (point - origin)
479        if local[2] <= 0:
480            return False
481
482        angle_x = np.degrees(np.arctan2(abs(local[0]), local[2]))
483        angle_y = np.degrees(np.arctan2(abs(local[1]), local[2]))
484        in_fov = (angle_x < (fov_x / 2) * margin) and (angle_y < (fov_y / 2) * margin)
485        if not in_fov:
486            return False
487
488        if depth_img is None:
489            return True
490
491        px = self.pos_calculator.pos_to_pixel(point[0], point[1], point[2], rotM, origin)
492        px_x, px_y = int(px[0]), int(px[1])
493        if px_x < 0 or px_y < 0:
494            return False
495
496        height, width = depth_img.shape[:2]
497        if px_x >= width or px_y >= height:
498            return False
499
500        depth_val = float(depth_img[px_y, px_x])
501        if depth_val <= 0 or np.isnan(depth_val):
502            return True
503
504        return local[2] <= depth_val + float(depth_tolerance)

506    def convert_graph_pos_to_pixel(self, graph, tx, rotM):
507        """Convert node 3D positions to normalized pixel coordinates.
508
509        Args:
510            graph: Graph dictionary with node positions.
511            tx: Translation vector.
512            rotM: Rotation matrix.
513
514        Returns:
515            Updated graph dictionary with x/y fields.
516        """
517        if graph is None:
518            return {"nodes": [], "edges": []}
519
520        height, width = img_size
521
522        for node in graph["nodes"]:
523            pos = node["pos"]
524            px = self.pos_calculator.pos_to_pixel(pos[0], pos[1], pos[2], rotM, tx)
525            x_pix, y_pix = float(px[0]), float(px[1])
526
527            # Clamp to pixel bounds
528            x_clamped = max(0, min(x_pix, width - 1))
529            y_clamped = max(0, min(y_pix, height - 1))
530
531            # GT: true if no clamping occurred (compare in pixel space)
532            gt_flag = (x_pix == x_clamped) and (y_pix == y_clamped)
533
534            # Convert to relative [0,1] and limit precision for VLM input
535            x_rel = round(x_clamped / (width - 1), 2)
536            y_rel = round(y_clamped / (height - 1), 2)
537
538            del node["pos"]
539            node["x"] = x_rel
540            node["y"] = y_rel
541            node["gt"] = gt_flag
542
543        return graph

545    def convert_graph_pixel_to_pos(self, graph, depth_img, tx, rotM):
546        """Convert node pixel coordinates to 3D positions using depth.
547
548        Args:
549            graph: Graph dictionary with node pixel coords.
550            depth_img: Depth image in meters.
551            tx: Translation vector.
552            rotM: Rotation matrix.
553
554        Returns:
555            Updated graph dictionary with pos fields.
556        """
557        height, width = img_size
558
559        for node in graph["nodes"]:
560            x_rel = node.get("x")
561            y_rel = node.get("y")
562
563            x_pix = int(round(x_rel * (width - 1)))
564            y_pix = int(round(y_rel * (height - 1)))
565
566            x_clipped = max(0, min(x_pix, width - 1))
567            y_clipped = max(0, min(y_pix, height - 1))
568
569            if x_pix != x_clipped or y_pix != y_clipped:
570                node["gt"] = False
571
572            pos = self.pos_calculator.pixel_to_pos(x_clipped, y_clipped, depth_img, rotM, tx)
573
574            del node["x"]
575            del node["y"]
576            node["pos"] = pos
577
578        return graph

280                    self.scene_graph[src][dst]["distance"] = distance
281                else:
282                    # New edge with appearance = 1
283                    self.scene_graph.add_edge(src, dst, rel=rel, distance=distance, appearance=1)

287        for edge in cutout_before.get("edges", []):
288            src, dst = edge["from"], edge["to"]
289            cutout_edge_set.add((src, dst))
290
291        # Decrease appearance for edges in cutout but NOT in VLM output
292        for src, dst in cutout_edge_set:
293            if (src, dst) not in vlm_edge_set:
294                if self.scene_graph.has_edge(src, dst):
295                    current_count = self.scene_graph[src][dst].get("appearance", 1)
296                    new_count = current_count - 1
297                    
298                    if new_count <= appearance_threshold:
299                        self.scene_graph.remove_edge(src, dst)
300                    else:
301                        self.scene_graph[src][dst]["appearance"] = new_count

457        """Check if a 3D point is within the camera's field of view and range.
458
459        Args:
460            point: 3D point in world coordinates.
461            origin: Camera origin in world coordinates.
462            rotM: Rotation matrix (camera to world).
463            depth_img: Optional depth image in meters for range cutoff.
464            depth_tolerance: Extra range beyond depth values (meters).
465
466        Returns:
467            True if point is likely visible, else False.
468        """
469        point = np.asarray(point, dtype=float)
470        origin = np.asarray(origin, dtype=float)
471
472        fx, fy = K_rgb[0, 0], K_rgb[1, 1]
473        height, width = img_size  
474
475        fov_x = 2 * np.degrees(np.arctan(width / (2 * fx)))
476        fov_y = 2 * np.degrees(np.arctan(height / (2 * fy)))
477
478        local = rotM.T @ (point - origin)
479        if local[2] <= 0:
480            return False
481
482        angle_x = np.degrees(np.arctan2(abs(local[0]), local[2]))
483        angle_y = np.degrees(np.arctan2(abs(local[1]), local[2]))
484        in_fov = (angle_x < (fov_x / 2) * margin) and (angle_y < (fov_y / 2) * margin)
485        if not in_fov:
486            return False
487
488        if depth_img is None:
489            return True
490
491        px = self.pos_calculator.pos_to_pixel(point[0], point[1], point[2], rotM, origin)
492        px_x, px_y = int(px[0]), int(px[1])
493        if px_x < 0 or px_y < 0:
494            return False
495

497        if px_x >= width or px_y >= height:
